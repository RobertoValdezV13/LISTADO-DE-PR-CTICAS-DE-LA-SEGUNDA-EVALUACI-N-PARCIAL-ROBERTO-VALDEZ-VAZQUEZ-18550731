print("Introduce tu Calificacion")
iCalifa = int(input())
if iCalifa == 100:
    print("A")
else:
    if iCalifa == 90:
        print("B")
    else:
        if iCalifa == 80:
            print("C")
        else:
            if iCalifa == 70:
                print("D")
            else:
                if iCalifa == 60:
                    print("F")
                else:
                    print("Calificacion fuera de rango")
